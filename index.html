async function loadRemoteSchedule(){ try{ const url = "https://sheetdb.io/api/v1/x97irced8j3de"; let res = await fetch(url); if(!res.ok) throw new Error("network error"); let data = await res.json(); let html = ""; (data||[]).forEach(row=>{ if(row.schedule){ html += `
${escapeHtml(row.schedule)}
`; } }); document.getElementById("scheduleList").innerHTML = html; }catch(e){ // fallback: local storage 일정 사용 loadSchedule(); } } function showPage(page){ document.querySelectorAll(".page").forEach(p=>{ p.classList.remove("active"); }); document.getElementById(page).classList.add("active"); if(page===\"schedule\"){ loadRemoteSchedule(); } }
