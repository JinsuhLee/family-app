function loadRemoteSchedule() { fetch('https://sheetdb.io/api/v1/x97irced8j3de') .then(response => response.json()) .then(data => { const scheduleList = document.getElementById('scheduleList'); if (!scheduleList) return; let html = ''; if (Array.isArray(data) && data.length > 0) { data.forEach(schedule => { html += `
${schedule.date || ''} ${schedule.time || ''}
${schedule.plan || ''}
`; }); } if (!html) html = "
일정 없음
"; scheduleList.innerHTML = html; }) .catch(() => { const scheduleList = document.getElementById('scheduleList'); if (scheduleList) scheduleList.innerHTML = "
일정 없음
"; }); } function showPage(page){ document.querySelectorAll('.page').forEach(p => { p.classList.remove('active'); }); document.getElementById(page).classList.add('active'); if (page === 'schedule') { loadRemoteSchedule(); } }
